BEGIN
    DECLARE v_user_id   BIGINT UNSIGNED;
    DECLARE v_apellido  VARCHAR(200);
    DECLARE v_existe    INT DEFAULT 0;

    -- Verificar email duplicado
    SELECT COUNT(*) INTO v_existe
    FROM users
    WHERE email = p_email;

    IF v_existe = 0 THEN

        SET v_apellido = TRIM(CONCAT(p_ap_paterno, ' ', p_ap_materno));

        INSERT INTO users (name, apellido, email, password, rol, created_at, updated_at)
        VALUES (
            TRIM(p_nombres),
            v_apellido,
            p_email,
            p_password_hash,
            'alumno',
            NOW(),
            NOW()
        );

        SET v_user_id = LAST_INSERT_ID();

        INSERT INTO alumnos (user_id, rut, nombre, apellido, nivel_actual, created_at, updated_at)
        VALUES (
            v_user_id,
            p_rut,
            TRIM(p_nombres),
            v_apellido,
            '3',
            NOW(),
            NOW()
        );

    END IF;
END
